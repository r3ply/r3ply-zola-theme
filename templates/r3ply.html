{% import "comments.html" as comments %} {# This macro takes a comment page, and will return a stringified list of ancestors comment ID's, ordered from most to least recent. It can be made into a real list by splitting on ',' `stop_at` is an optional parameter, where you can specify the ID of an ancestor to stop at. Note: how everything is printed matters so be careful with whitespace E.g. A \ B \ C get_ancestors(C) -> "[C, B, A]" #} {%- macro get_ancestors(node, stop_at=false) -%} {{ node.slug }}{# always print the current node #} {%- set ancestor = node.extra.comment.in_reply_to -%} {%- if ancestor == false or stop_at | as_str in ancestor -%}{# base case: ancestor of node is false or end—do nothing #} {%- else -%}{# recursive case: print comma and call get_root, continuing upwards #} {%- set ancestor=get_taxonomy_term(kind="threads", term=node.taxonomies.replies[1]) | get(key="pages") | nth(n=0) -%},{{ self::get_ancestors(node=ancestor, stop_at=stop_at) }} {%- endif -%} {%- endmacro -%} {# This macro takes a list of comments as zola pages and it renders them in the order of thread with the most recent activity. A thread is a comment and all of its descendants. set_global list_with_activity = [] %} {% for list in list_of_lists_of_ancestors %} {% for node in list %} {# Get latest activity among ancestors #} {% if loop.first %}{% set_global latest_activity =
